ORG 0x600
WORD 100
WORD 228
WORD 200

ORG 0x700
little_byte: WORD ?
big_byte: WORD ?
current: WORD 0x600
len: WORD 15
temp: WORD ?
sign_mask: WORD 0x0400
mask: WORD 0x07FF
minus_mask: WORD 0xF800
sup: WORD 500

START:
	CLA
	ST little_byte
	ST big_byte
number:
	LD #18
	SUB len
	AND #3
	BNE halal
	LD (current)+
	JUMP outro
halal:
	LD (current)+
	AND mask
	ST $temp
	AND sign_mask
	BEQ plus
	JUMP minus
plus:
	LD $temp
	CALL function
	JUMP outro
minus:
	LD $temp
	OR minus_mask
	ST $temp
	CALL function
	JUMP outro
outro:
	LOOP len
	HLT
	JUMP number

function:
	ASL
	ADD $temp
	ADD #200
	CMP sup
	BGE over
	JUMP aboba
over:
	LD sup
aboba:
	ADD little_byte
	ST little_byte
	CLA
	ADC big_byte
	ST big_byte
	RET
