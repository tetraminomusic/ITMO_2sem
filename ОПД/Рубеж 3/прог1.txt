ORG 0x228
current: WORD 0x400
len: WORD 12

START:
	CLA
	ST $sum
number:
	LD #12
	SUB len
	AND #1
	BEQ eval
	LD (current)+
	JUMP outro
eval:
	LD (current)+
	CALL function
	JUMP outro
outro:
	LOOP len
	JUMP number
	HLT

function:
	ASL
	SUB #100
	BMI return
	ADD $sum
	ST $sum
	return: RET
